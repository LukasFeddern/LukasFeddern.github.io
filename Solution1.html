7) Recode all emtries of “Currently have” and “Had, within 9 months” into “yes”

  • R
  • Stata

Note that options 3 and 4 use across(). This command specifies that you want to apply a function across specific columns.

Option 1

peru[peru=="Had, within 9 months"|peru=="Currently have"] <- "yes"

Option 2

peru <- mutate(peru, 
fever = ifelse(fever=="Had, within 9 months"|fever=="Currently have", "yes", fever),
nasalcongestion = ifelse(nasalcongestion=="Had, within 9 months"|nasalcongestion=="Currently have", "yes", nasalcongestion),
breathdifficulty = ifelse(breathdifficulty=="Had, within 9 months"|breathdifficulty=="Currently have", "yes", breathdifficulty),
generaldiscomfort = ifelse(generaldiscomfort=="Had, within 9 months"|generaldiscomfort=="Currently have", "yes", generaldiscomfort),
diarrhea = ifelse(diarrhea=="Had, within 9 months"|diarrhea=="Currently have", "yes", diarrhea),
nausea = ifelse(nausea=="Had, within 9 months"|nausea=="Currently have", "yes", nausea),
cefalea = ifelse(cefalea=="Had, within 9 months"|cefalea=="Currently have", "yes", cefalea),
cough = ifelse(cough=="Had, within 9 months"|cough=="Currently have", "yes", cough),
sorethroat = ifelse(sorethroat=="Had, within 9 months"|sorethroat=="Currently have", "yes", sorethroat),
symptoms3cat = ifelse(sorethroat=="Had, within 9 months"|sorethroat=="Currently have", "yes", symptoms3cat)
)

Option 3

peru <- mutate(peru, 
across(.cols = c(fever:sorethroat, symptoms3cat),
~case_when(.=="Had, within 9 months"| .=="Currently have"~"yes",
!(.=="Had, within 9 months"| .=="Currently have") ~ .)))

Option 4

peru <- mutate(peru, 
across(.cols = fever:sorethroat,
~recode(.x, "Had, within 9 months" = "yes", "Currently have" = "yes")))
replace fever = "yes" if fever = "Had, within 9 months"|fever="Currently have"
